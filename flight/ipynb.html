def cache_airports_delays(dat):
    delay_agg_airport = {}

    res = dat.ArrDelay.groupby(dat.Origin).mean().reset_index()
    for i in xrange(len(res)):
        delay_agg_airport[('arr', 'origin', res.Origin[i])] = res.ArrDelay[i]

    res = dat.ArrDelay.groupby(dat.Dest).mean().reset_index()
    for i in xrange(len(res)):
        delay_agg_airport[('arr', 'dest', res.Dest[i])] = res.ArrDelay[i]

    res = dat.DepDelay.groupby(dat.Origin).mean().reset_index()
    for i in xrange(len(res)):
        delay_agg_airport[('dep', 'origin', res.Origin[i])] = res.DepDelay[i]

    res = dat.DepDelay.groupby(dat.Dest).mean().reset_index()
    for i in xrange(len(res)):
        delay_agg_airport[('dep', 'dest', res.Dest[i])] = res.DepDelay[i]

    return delay_agg_airport

def routes_delays(dat):
    res = dat.groupby(['Origin','Dest']).ArrDelay.mean().reset_index()
    cache = {}
    for i in xrange(len(res)):
        cache[(res.Origin[i], res.Dest[i])] = res.ArrDelay[i]    
    return cache

def carrier_delays(dat):
    res = dat.groupby(['UniqueCarrier']).ArrDelay.mean().reset_index()
    cache = {}
    for i in xrange(len(res)):
        cache[res.UniqueCarrier[i]] = res.ArrDelay[i]    
    return cache

def routes_delays(dat):
    res = dat.groupby(['Origin','Dest']).ArrDelay.mean().reset_index()
    cache = {}

def carrier_delays(dat):

    return cache
